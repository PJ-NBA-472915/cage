```json
{
  "method": "tools/call",
  "tool": "agent_create",
  "arguments": {
    "name": "Python Planner",
    "role": "planner",
    "config": {}
  }
}
```

### 2. Create Crew with Agent UUIDs
```json
{
  "method": "tools/call",
  "tool": "crew_create",
  "arguments": {
    "name": "Dev Crew",
    "roles": {
      "planner": "agent-uuid-1",
      "implementer": "agent-uuid-2"
    }
  }
}
```

### 3. Run Crew with TaskSpec
```json
{
  "method": "tools/call",
  "tool": "crew_run",
  "arguments": {
    "crew_id": "crew-uuid",
    "task": {
      "title": "Build Feature",
      "description": "Implement the feature",
      "acceptance": [
        "Feature works correctly",
        "Tests pass"
      ]
    }
  }
}
```
